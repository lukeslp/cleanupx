# These provide foundational support for progress tracking and event emissions.

class EventEmitter:
    """Event emitter for progress tracking"""
    def __init__(self, event_emitter: Callable[[dict], Any] = None):
        self.event_emitter = event_emitter

    async def progress_update(self, description):
        await self.emit(description)

    async def error_update(self, description):
        await self.emit(description, "error", True)

    async def success_update(self, description):
        await self.emit(description, "success", True)

    async def emit(self, description="Unknown State", status="in_progress", done=False):
        if self.event_emitter:
            await self.event_emitter({
                "type": "status",
                "data": {
                    "status": status,
                    "description": description,
                    "done": done,
                },
            })

def find_server_processes():
    """Find all running MoE server processes."""
    try:
        cmd = "ps aux | grep 'python.*server.py' | grep -v grep"
        output = subprocess.check_output(cmd, shell=True).decode()
        processes = []
        for line in output.splitlines():
            parts = line.split()
            pid = int(parts[1])
            cmd_line = ' '.join(parts[10:])
            if any(script in cmd_line for script in ['caminaa_server.py', 'belters_server.py']):
                processes.append({"pid": pid, "command": cmd_line})
        return processes
    except subprocess.CalledProcessError as e:
        logger.error(f"Error finding processes: {e}")
        return []
